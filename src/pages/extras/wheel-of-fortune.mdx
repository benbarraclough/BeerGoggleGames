<script is:inline>
  (() => {
    // Placeholder data (lightweight). Replace with real data or dynamic fetch later.
    const categories = {
      "Card Games": ["Blow Me","Cross the Bridge","Curse","Dick Head","Kings"],
      "Pong Games": ["Beer Pong","Chandelier","Baseball","Blow Ball"],
      "Cup Games": ["Flip Cup","Get Stacked","Blow Cup","Rage Cage"],
      "Dice Games": ["7 11 Doubles","Beer Die","Mexican","Three Man"],
      "Coin Games": ["Arrogance","Hockey","Doom","Quarters"],
      "Vocal Games": ["21","Around the World","Black Black White","Don't Interrupt"],
      "Misc Games": ["Battle Shots","Deceit","Drunk Jenga","Fingers Ony"],
      "Outdoor Games": ["Beer Darts","Dizzy Bat","Flunky Ball","Detonator"]
    };

    const allGames = Object.values(categories).flat();
    const catBtn = document.getElementById('spin-category');
    const catResult = document.getElementById('category-result');
    const gameBtn = document.getElementById('spin-game');
    const gameResult = document.getElementById('game-result');
    const filter = document.getElementById('game-category-filter');

    if (filter) {
      Object.keys(categories).forEach(c => {
        const opt = document.createElement('option');
        opt.value = c;
        opt.textContent = c;
        filter.appendChild(opt);
      });
    }

    function randomItem(arr) {
      return arr[Math.floor(Math.random() * arr.length)];
    }

    catBtn?.addEventListener('click', () => {
      const keys = Object.keys(categories);
      const picked = randomItem(keys);
      catResult.textContent = picked;
      catResult.classList.add('animate-pulse');
      setTimeout(() => catResult.classList.remove('animate-pulse'), 600);
    });

    gameBtn?.addEventListener('click', () => {
      const cat = filter?.value || '';
      const pool = cat && categories[cat] ? categories[cat] : allGames;
      const picked = randomItem(pool);
      gameResult.textContent = picked;
      gameResult.classList.add('animate-pulse');
      setTimeout(() => gameResult.classList.remove('animate-pulse'), 600);
    });
  })();
</script>
