  <p><strong>Origin:</strong> Italy</p>

<DrinkSection title="Ingredients" icon="/images/liquor.webp">
  <ul>
  <li>Aperol - 100ml</li>
  <li>Prosecco - 150ml</li>
  <li>Soda water - 50ml</li>
  <li>Orange - 1 Slice</li>
  <li>Ice</li>
</ul>
</DrinkSection>

<DrinkSection title="Recipe" icon="/images/rules.webp">
  <ol>
  <li>Add a handful of ice to a glass</li>
  <li>Pour the aperol and prosecco into the glass and top up with soda</li>
  <li>Cut a slice of orange and add to the drink to finish</li>
</ol>
</DrinkSection>

<DrinkSection title="Tips" icon="/images/lightbulb.webp">
  <ul>
    <li>Use aperol glasses to make the drink fully authentic.</li>
    <li>Use orange bitters to add another element to the drink</li>
  </ul>
</DrinkSection>
