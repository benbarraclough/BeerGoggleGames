<DrinkSection title="Ingredients" icon="/BeerGoggleGames/images/liquor.webp">
  <ol>
  <li>Vodka - 50ml</li>
  <li>Peach Schnapps - 25ml</li>
  <li>Peach syrup - 20ml</li>
  <li>Cranberry juice - 100ml</li>
  <li>Lime juice - 20ml</li>
  <li>Lime - 1 wedge</li>
  <li>Ice - 1 handful</li>
</ol>
</DrinkSection>

<DrinkSection title="Recipe" icon="/BeerGoggleGames/images/rules.webp">
  <ol>
  <li>Add the vodka, peach schnapps, peach syrup, cranberry juice and lime juice to a shaker with ice.</li>
  <li>Shake the shaker until ice cold.</li>
  <li>Pour the drink into a glass with the ice.</li>
  <li>Garnish with a wedge of lime and serve.</li>
</ol>
</DrinkSection>
